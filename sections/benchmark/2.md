```html
<script>
import {shuffle} from 'lodash'
var total = 500
var items = []
for (var i = 0; i < total; i++) {
  items.push({
    id: i,
    label: String(Math.random()).slice(0, 5)
  })
}
let s = window.performance.now()

export default{
    data(){
        return {
        total: total,
        time: 0,
        action: 'Render',
        items: items,
        selected: null,
        actions:['删除','更新']
        }
    },
    mounted(){
        this.time = window.performance.now() - s
    },
  methods: {
    shuffle(){
      this.action = 'shuffle'
      this.items = shuffle(this.items)
      let s = window.performance.now()
      this.$nextTick(()=>{
        this.time = window.performance.now() - s
      })
    }
  }
}

</script>
```

### 官方解析
#### 性能相关
* Client

1. Proxy vs defineProperties
2. 优化 - `diff` （尤：静态分析）

* SSR

1. 重写渲染器,静态字符串直接字符串拼接
2. 最大化异步，更好处理并发，动静比越大优势越大

#### 其他

1. modularization

每个模块均可独立引用

* `customRender`
* `Tree shaking`

2. Composition 

### Proxy vs defineProperties

客户端渲染首屏提升过于明显

```javascript
const Benchmark = require("benchmark");

const suite = new Benchmark.Suite();

const proxyCreate = () => {
    const _obj = {};
    const proxy = new Proxy(_obj, {
        set: (obj, prop, value) => {
            _obj[prop] = value;
        },
        get: (obj, prop, receiver) => Reflect.get(obj, prop, receiver),
    });
    return proxy;
};

const defineCreate = () => {
    const defineProp = {
      a: 'a',
      b: 'b',
      c: 'c',

      d: 'd',
      e: 'e',
      f: 'f',

      g: 'g',
      h: 'h',
      i: 'i'
    };

    const define = (obj, key) => {
        Object.defineProperty(obj, key, {
            configurable: true,
            enumerable: true,
            set: (v) => (defineProp._v = v),
            get: () => defineProp._v,
        });
    };

    Object.keys(defineProp).forEach((key) => {
        define(defineProp, key);
    });

    return defineProp;
};

suite
    .add("proxy", function () {
        // proxy.prop = 5;
        // proxy.prop;
        proxyCreate();
    })
    .add("defineProperty", function () {
        // defineProp.prop = 5;
        // defineProp.prop;
        defineCreate();
    })
    .on("cycle", function (event) {
        console.log(String(event.target));
    })
    .on("complete", function () {
        console.log("Fastest is " + this.filter("fastest").map("name"));
    })
    .run();
```
